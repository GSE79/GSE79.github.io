65  MCU2Port.enQueueOutPacket(VERSION, packType_ReadComplete);

76  {
77  if (PDUPort.getOutPackQueueDepth() == 0)
78  PDUPort.enQueueOutPacket(PDU, packType_WriteComplete);

688 {

693  break;
694  case NACK: //
695  break;
696  case PING: //
697  break;

705  {

716  }
717  else if (SNMP_CurrentState == WAITING_FOR_OFFSET)
718  {
719  //convert string to offset
720  SNMP_CurrentOffset = (int)strtol(SNMP_ReadBuffer, NULL, 10);
721 
722  //Debug messages
723  DebugPRINT("Offset to set: ");
724  DebugPRINTLN(SNMP_CurrentOffset);
725 
726  SNMP_CurrentState = WAITING_FOR_DATA;
727  SNMP_StateSustain = true;
728 
729  //Flush buffer and reset buffer index
730  SNMP_ReadBuffer_Index = 0;
731  memset(SNMP_ReadBuffer, 0, MAX_OID_LENGTH);
732  return -1;
733  }
734  else if (SNMP_CurrentState == WAITING_FOR_DATA)
735  {
736  //Debug messages
737  DebugPRINT("Old value in Node Array: ");
738  DebugPRINTLN(node_array[SNMP_CurrentOffset].value);
739 
740  strncpy(node_array[SNMP_CurrentOffset].value, SNMP_ReadBuffer, MAX_OID_LENGTH);
741 
742  node_array[SNMP_CurrentOffset].changed = true;
743 
744  //Debug messages
745  DebugPRINT("New value in Node Array: ");
746  DebugPRINTLN(node_array[SNMP_CurrentOffset].value);
747 
748  SNMP_CurrentState = IDLE;
749 
750  //Flush buffer and reset buffer index
751  SNMP_ReadBuffer_Index = 0;
752  memset(SNMP_ReadBuffer, 0, MAX_OID_LENGTH);
753  return SNMP_CurrentOffset;
754  }
755  }
756  break;
757  }
758 
759  SNMP_ReadBuffer_Index = 0;
760  memset(SNMP_ReadBuffer, 0, MAX_OID_LENGTH);
761  SNMP_StateSustain = false;
762  return -1;

768 void ACU_INO::CommitChangesToMiB()
769 {
770  static short LED_Color;
771  for (int i = 0; i < END_OF_NODES; i++)
772  {
773 
774  if (node_array[i].changed == 2)
775  {
776  node_transport_struct sender = { MIB_OID_DATA, i };
777 
778  //Send Up to NVIDIA
779  SendChangeToMnC(&sender, node_array[i].value);
780 
781  //Wait for Acknowledgement
782  int CharRead = -1;
783  while (node_array[i].changed != 0)
784  {
785  #define ReadTimeout 250
786  unsigned long currMillis = millis();
787  static unsigned long prevMillis = currMillis;
788 
789  do {
790  char inChar = GetLCDMenuChar();
791  if (inChar != 0x00 && inChar != ASCII_cr && inChar != ASCII_lf)
792  {
793  DebugPRINTLN("SCREEN ON!");
794  ChangeLCDBrightDim(true);
795  return;
796  }
797  currMillis = millis();
798  CharRead = NVIDIASerial.read();
799 
800  if (currMillis - prevMillis > ReadTimeout)
801  {
802 
803  if (LED_Color == LED_OFF)
804  LED_Color = LED_YELLOW;
805  else
806  LED_Color = LED_OFF;
807 
808  prevMillis = currMillis;
809  break;
810  }
811  } while (!(CharRead == 32 || CharRead == 33));//(CharRead == -1 || CharRead >= 127);
812 
813  if (CharRead == ACK)
814  {
815  node_array[i].changed = 0;
816  DebugPRINTLN("ACK\nFlushing Buffer...");
817  NVIDIASerial.flush();
818  break;
819  }
820  else if (CharRead == NACK)
821  {
822  DebugPRINTLN("NACK...Resending...");
823  SendChangeToMnC(&sender, node_array[i].value);
824  DebugPRINTLN("Flushing Buffer...");
825  NVIDIASerial.flush();
826  }
827  else
828  {
829  DebugPRINTLN("Resending...");
830  SendChangeToMnC(&sender, node_array[i].value);
831  }
832  }
833  }
834  }
835  TriggerNVIDIAUpdate = false;
836 }
